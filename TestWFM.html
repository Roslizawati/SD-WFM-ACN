<script>

  const daysInMonth = 30;


  function generateSchedule() {

    const scheduleGrid = document.getElementById("schedule-grid");

    scheduleGrid.innerHTML = "";


    // Header row

    scheduleGrid.innerHTML += `<div class="header">Agent / Day</div>`;

    for (let h = 0; h < 24; h++) {

      scheduleGrid.innerHTML += `<div class="header">${h.toString().padStart(2, '0')}:00</div>`;

    }


    const selected = document.getElementById("agent1-schedule").value;

    let [start1, end1] = selected.split('-').map(Number);

    const start2 = 10, end2 = 19;


    for (let day = 1; day <= daysInMonth; day++) {

      // Agent 1 row

      scheduleGrid.innerHTML += `<div class="agent-day">Agent 1 - Day ${day}</div>`;

      for (let h = 0; h < 24; h++) {

        if (h >= start1 && h < end1) {

          scheduleGrid.innerHTML += `<div class="shift">W</div>`;

        } else {

          scheduleGrid.innerHTML += `<div class="empty"></div>`;

        }

      }


      // Agent 2 row (fixed shift)

      scheduleGrid.innerHTML += `<div class="agent-day">Agent 2 - Day ${day}</div>`;

      for (let h = 0; h < 24; h++) {

        if (h >= start2 && h < end2) {

          scheduleGrid.innerHTML += `<div class="shift">W</div>`;

        } else {

          scheduleGrid.innerHTML += `<div class="empty"></div>`;

        }

      }

    }

  }

</script>
